📊 项目1：每日数据报表自动化

需求描述

每天早上8点，自动读取销售数据CSV文件，生成包含各部门汇总、TOP10客户、同比环比的Excel报表，发送给管理层邮件。

python — daily_report.py
""" 项目1：每日销售报表自动化 功能：读取CSV → 数据处理 → 生成格式化Excel → 发送邮件 """ import pandas as pd import smtplib import schedule import time import os from email.mime.multipart import MIMEMultipart from email.mime.text import MIMEText from email.mime.base import MIMEBase from email import encoders from datetime import datetime, timedelta from openpyxl.styles import Font, PatternFill, Alignment from openpyxl.utils import get_column_letter from loguru import logger from dotenv import load_dotenv load_dotenv() # ===== 配置 ===== EMAIL_HOST = "smtp.qq.com" EMAIL_USER = os.getenv("EMAIL_USER") EMAIL_PASS = os.getenv("EMAIL_PASS") RECIPIENTS = ["manager@company.com", "ceo@company.com"] def generate_report(data_file: str) -> str: """生成格式化Excel报表，返回文件路径""" logger.info(f"读取数据文件：{data_file}") df = pd.read_csv(data_file, encoding="utf-8") # 数据清洗 df["日期"] = pd.to_datetime(df["日期"]) df["金额"] = pd.to_numeric(df["金额"], errors="coerce").fillna(0) df = df.drop_duplicates(subset=["订单号"]) # 获取今天和昨天的数据 today = datetime.now().date() yesterday = today - timedelta(days=1) df_today = df[df["日期"].dt.date == yesterday] # 取昨天数据（今天早上跑） # 部门汇总 dept_summary = df_today.groupby("部门").agg({ "金额": ["sum", "mean", "count"] }).round(2) dept_summary.columns = ["销售总额", "平均单价", "订单数量"] dept_summary = dept_summary.sort_values("销售总额", ascending=False) # TOP10客户 top_customers = df_today.groupby("客户名")["金额"].sum().sort_values(ascending=False).head(10) # 生成Excel filename = f"销售日报_{yesterday.strftime('%Y%m%d')}.xlsx" with pd.ExcelWriter(filename, engine="openpyxl") as writer: df_today.to_excel(writer, sheet_name="明细数据", index=False) dept_summary.to_excel(writer, sheet_name="部门汇总") top_customers.to_excel(writer, sheet_name="TOP10客户") # 美化表头 for sheet_name in writer.sheets: ws = writer.sheets[sheet_name] for cell in ws[1]: cell.font = Font(bold=True, color="FFFFFF") cell.fill = PatternFill(start_color="1F4E79", fill_type="solid") logger.success(f"✅ 报表已生成：{filename}") return filename def send_email(subject: str, body: str, attachment: str): """发送带附件的邮件""" msg = MIMEMultipart() msg["From"] = EMAIL_USER msg["To"] = ", ".join(RECIPIENTS) msg["Subject"] = subject msg.attach(MIMEText(body, "html", "utf-8")) # 添加附件 with open(attachment, "rb") as f: part = MIMEBase("application", "octet-stream") part.set_payload(f.read()) encoders.encode_base64(part) part.add_header("Content-Disposition", f"attachment; filename={os.path.basename(attachment)}") msg.attach(part) with smtplib.SMTP_SSL(EMAIL_HOST, 465) as server: server.login(EMAIL_USER, EMAIL_PASS) server.sendmail(EMAIL_USER, RECIPIENTS, msg.as_string()) logger.success("✅ 邮件发送成功") def daily_job(): try: report_file = generate_report("./data/sales.csv") email_body = f"""

销售日报 - {datetime.now().strftime('%Y年%m月%d日')}

请查收今日销售报表，详见附件。

如有问题请联系数据组。

""" send_email(f"销售日报 {datetime.now().strftime('%m/%d')}", email_body, report_file) except Exception as e: logger.exception("日报任务失败") # 定时执行 schedule.every().day.at("08:00").do(daily_job) if __name__ == "__main__": daily_job() # 先手动运行一次测试 while True: schedule.run_pending() time.sleep(60)

🧠 项目3：AI智能工单处理系统

需求描述

收到业务部门的工单邮件 → AI自动分类和提取信息 → 查询知识库找解决方案 → 生成标准化回复 → 自动发送回复邮件。这是最贴合岗位需求的项目！

python — ticket_handler.py
""" 项目3：AI智能工单处理系统 技术栈：OpenAI/DeepSeek API + Python """ import json from openai import OpenAI from loguru import logger from datetime import datetime client = OpenAI( api_key="your-deepseek-key", base_url="https://api.deepseek.com/v1" ) # 知识库（实际项目可以从数据库或文件读取） KNOWLEDGE_BASE = { "账号问题": "账号相关问题请联系IT部门，邮箱：it@company.com，处理时间1个工作日", "数据问题": "数据问题请提供：数据表名、时间范围、具体异常描述，数据组会在2小时内响应", "报表问题": "报表显示异常请截图，并说明报表名称和查看时间，BI团队工作日内处理", "系统故障": "系统故障请立即拨打运维热线：xxxx，同时提交此工单记录", } def classify_ticket(content: str) -> dict: """AI分类工单内容""" prompt = f""" 你是一个工单分类系统。请分析以下工单内容，提取关键信息。 工单内容： {content} 请以JSON格式输出（不要有其他文字）： {{ "category": "工单类别（账号问题/数据问题/报表问题/系统故障/其他）", "priority": "优先级（高/中/低）", "requester": "申请人（从内容推断，不确定填未知）", "summary": "问题摘要（50字以内）", "key_info": "关键信息提取（如涉及的系统、时间、数据范围等）" }} """ response = client.chat.completions.create( model="deepseek-chat", messages=[{"role": "user", "content": prompt}], temperature=0 ) try: return json.loads(response.choices[0].message.content) except: return {"category": "其他", "priority": "中", "summary": content[:50]} def generate_reply(ticket_content: str, classify_result: dict) -> str: """AI生成标准化回复""" category = classify_result.get("category", "其他") kb_answer = KNOWLEDGE_BASE.get(category, "感谢您的反馈，我们会尽快处理。") prompt = f""" 你是一个专业的客户服务人员，请根据工单内容和处理规范生成一封礼貌的回复邮件。 原始工单： {ticket_content} 工单分类：{category} 优先级：{classify_result.get('priority', '中')} 问题摘要：{classify_result.get('summary', '')} 处理规范（参考，不要照抄）： {kb_answer} 要求： 1. 语气专业友好 2. 确认问题已收到 3. 说明处理方式和预计时间 4. 如需提供更多信息，明确说明需要什么 5. 用中文，格式清晰 """ response = client.chat.completions.create( model="deepseek-chat", messages=[{"role": "user", "content": prompt}], temperature=0.7 ) return response.choices[0].message.content def process_ticket(ticket_content: str) -> dict: """完整的工单处理流程""" logger.info("开始处理工单...") # 1. 分类提取 classify_result = classify_ticket(ticket_content) logger.info(f"工单分类：{classify_result}") # 2. 生成回复 reply = generate_reply(ticket_content, classify_result) # 3. 保存记录 record = { "ticket_id": f"TK-{datetime.now().strftime('%Y%m%d%H%M%S')}", "content": ticket_content, "classify": classify_result, "reply": reply, "created_at": datetime.now().isoformat() } # 实际项目：保存到数据库 # save_to_db(record) logger.success(f"✅ 工单 {record['ticket_id']} 处理完成") return record # 测试 if __name__ == "__main__": test_ticket = """ 您好，我是运营部的小李， 今天我在系统里查看上周的销售数据时，发现2024年1月10日到1月15日的数据显示不出来， 页面显示"数据加载失败"，但其他时间段都正常。 请帮忙排查一下，这些数据很重要，今天要汇报给总经理。 谢谢！ """ result = process_ticket(test_ticket) print("\n===== 工单分类 =====") print(json.dumps(result["classify"], ensure_ascii=False, indent=2)) print("\n===== AI生成回复 =====") print(result["reply"])
