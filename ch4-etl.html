Pandas 是什么？

Pandas是Python里处理表格数据的神器，可以理解为"会编程的超级Excel"。
核心数据结构：DataFrame（二维表格，有行有列）和 Series（一维列表）。

1

创建和读取数据

python
import pandas as pd import numpy as np # ===== 从字典创建DataFrame ===== df = pd.DataFrame({ "姓名": ["张三", "李四", "王五", "赵六"], "部门": ["财务", "运营", "财务", "技术"], "工资": [15000, 12000, 18000, 20000], "入职年份": [2020, 2022, 2019, 2021] }) # ===== 读取各种数据源 ===== df = pd.read_excel("数据.xlsx", sheet_name="Sheet1") # Excel df = pd.read_excel("数据.xlsx", sheet_name=0, skiprows=2) # 跳过前2行标题 df = pd.read_csv("数据.csv", encoding="utf-8") # CSV df = pd.read_csv("数据.csv", encoding="gbk") # 中文乱码时用gbk # 读取多个Excel Sheet合并 sheets = pd.read_excel("数据.xlsx", sheet_name=None) # 读取所有sheet，返回字典 df_all = pd.concat(sheets.values(), ignore_index=True) # 合并所有sheet # ===== 查看数据基本信息 ===== print(df.head(5)) # 前5行 print(df.tail(3)) # 后3行 print(df.shape) # (行数, 列数) print(df.columns.tolist()) # 所有列名 print(df.dtypes) # 每列的数据类型 print(df.info()) # 详细信息：列名、非空数量、类型 print(df.describe()) # 数值列的统计摘要 print(df.isnull().sum()) # 每列空值数量

3

数据清洗（最核心！）

数据清洗是ETL中工作量最大的部分，占总时间的60-80%

真实数据总是脏乱差：有空值、有重复、格式不统一、有异常值……必须清洗才能用。

python
# ===== 处理重复数据 ===== print(df.duplicated().sum()) # 查看重复行数量 df = df.drop_duplicates() # 删除完全重复的行 df = df.drop_duplicates(subset=["订单号"]) # 按指定列去重 df = df.drop_duplicates(subset=["订单号"], keep="last") # 保留最后一条 # ===== 处理空值 ===== df.isnull().sum() # 查看每列空值数 # 删除空值 df = df.dropna() # 删除任何列有空值的行 df = df.dropna(subset=["订单号", "客户名"]) # 只看关键列 df = df.dropna(thresh=3) # 至少3个非空值才保留 # 填充空值 df["工资"] = df["工资"].fillna(0) # 用0填充 df["部门"] = df["部门"].fillna("未知") # 用字符串填充 df["工资"] = df["工资"].fillna(df["工资"].mean()) # 用均值填充 df["工资"] = df["工资"].ffill() # 用前一个值填充（向前填充） # ===== 类型转换 ===== df["工资"] = pd.to_numeric(df["工资"], errors="coerce") # 非数字变NaN df["日期"] = pd.to_datetime(df["日期"], format="%Y/%m/%d") # 字符串转日期 df["工资"] = df["工资"].astype(int) # 转整数 df["工号"] = df["工号"].astype(str) # 转字符串 # ===== 字符串清洗 ===== df["姓名"] = df["姓名"].str.strip() # 去两端空格 df["部门"] = df["部门"].str.strip().str.upper() # 去空格并转大写 df["电话"] = df["电话"].str.replace("-", "") # 删除破折号 df["金额"] = df["金额"].str.replace("¥", "").str.replace(",", "") # 去货币符号 # ===== 异常值处理 ===== # 找出工资异常高的数据（超过3倍标准差） mean = df["工资"].mean() std = df["工资"].std() outliers = df[(df["工资"] > mean + 3 * std) | (df["工资"] < mean - 3 * std)] print(f"发现{len(outliers)}条异常数据") # 将异常值替换为边界值 df["工资"] = df["工资"].clip(3000, 100000) # 限定在3000-100000之间

🔄 4.2 完整ETL项目示例

python
""" 完整ETL示例：读取多个月度销售Excel → 合并清洗 → 按部门汇总 → 写入数据库 → 生成报表 """ import pandas as pd import glob import logging from sqlalchemy import create_engine from datetime import datetime logger = logging.getLogger("etl") logging.basicConfig(level=logging.INFO, format='%(asctime)s [%(levelname)s] %(message)s') def extract(data_dir: str) -> pd.DataFrame: """采集：读取目录下所有Excel文件并合并""" all_files = glob.glob(f"{data_dir}/*.xlsx") logger.info(f"找到 {len(all_files)} 个Excel文件") dfs = [] for file in all_files: try: df = pd.read_excel(file) df["来源文件"] = file # 标记数据来源 dfs.append(df) logger.info(f"✅ 读取 {file}：{len(df)} 行") except Exception as e: logger.error(f"❌ 读取失败 {file}：{e}") if not dfs: raise ValueError("没有成功读取任何文件") combined = pd.concat(dfs, ignore_index=True) logger.info(f"合并后共 {len(combined)} 行数据") return combined def transform(df: pd.DataFrame) -> pd.DataFrame: """转换：数据清洗和处理""" logger.info("开始数据清洗...") # 1. 标准化列名（去空格、统一命名） df.columns = df.columns.str.strip() # 2. 删除完全空白的行 df = df.dropna(how="all") # 3. 处理关键字段空值 df = df.dropna(subset=["订单号"]) # 订单号不能为空 df["金额"] = df["金额"].fillna(0) df["部门"] = df["部门"].fillna("未分类") # 4. 类型转换 df["金额"] = pd.to_numeric(df["金额"], errors="coerce").fillna(0) df["日期"] = pd.to_datetime(df["日期"], errors="coerce") # 5. 去重 before_count = len(df) df = df.drop_duplicates(subset=["订单号"]) logger.info(f"去重：删除 {before_count - len(df)} 条重复记录") # 6. 新增计算列 df["含税金额"] = (df["金额"] * 1.13).round(2) df["年份"] = df["日期"].dt.year df["月份"] = df["日期"].dt.month logger.info(f"清洗完成，剩余 {len(df)} 行有效数据") return df def load(df: pd.DataFrame, db_url: str): """加载：写入数据库并生成Excel报表""" # 写入数据库 engine = create_engine(db_url) df.to_sql("sales_data", engine, if_exists="append", index=False) logger.info(f"✅ 写入数据库：{len(df)} 条") # 生成汇总报表 summary = df.groupby(["部门", "月份"]).agg({ "金额": "sum", "订单号": "count" }).rename(columns={"订单号": "订单数"}).reset_index() filename = f"汇总报表_{datetime.now().strftime('%Y%m%d_%H%M')}.xlsx" summary.to_excel(filename, index=False) logger.info(f"✅ 报表已生成：{filename}") return filename # 主程序 if __name__ == "__main__": raw = extract("./data") clean = transform(raw) load(clean, "mysql+pymysql://user:pass@localhost/salesdb")
